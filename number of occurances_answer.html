class Solution {
    int count(int[] arr, int n, int x) {
        // code here
        int first = firstOccurence(arr, n, x);
        if(first == -1){
            return 0;
        }
        return (lastOccurence(arr, n, x) - first)+1;
    }
    
    int firstOccurence(int arr[], int n, int x){
        int low=0;
        int high = n-1;
        int result=-1;
        while(low <= high){
            int mid = (low+high)/2;
            if(arr[mid] == x){
                result = mid;
                high = mid-1;
            }
            else if(arr[mid] > x){
                high = mid-1;
            }
            else{
                low = mid+1;
            }
        }
        return result;
    }
    
    
    
     int lastOccurence(int arr[], int n, int x){
        int low=0;
        int high = n-1;
        int result=-1;
        while(low <= high){
            int mid = (low+high)/2;
            if(arr[mid] == x){
                result = mid;
                low = mid+1;
            }
            else if(arr[mid] > x){
                high = mid-1;
            }
            else{
                low = mid+1;
            }
        }
        return result;
    }
}
