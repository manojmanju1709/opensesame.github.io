class Solution:
    ##Complete this function
    def ternarySearch(self,arr,N,K):
        #Your code here
        q = []
        q.append(arr)
        while len(q) > 0:
            arr = q.pop()
            #print(arr)
            if len(arr) > 1:
                mid1 = len(arr) // 3
                mid2 = len(arr) * 2 // 3
                
                left = arr[:mid1]
                mid = arr[mid1:mid2]
                right = arr[mid2:]
                
                if K < arr[mid1] and len(left) > 0:
                    q.append(left)
                elif K < arr[mid2] and len(mid) > 0:
                    q.append(mid)
                elif len(right) > 0:
                    q.append(right)
            else:
                #print(arr)
                if arr[0] == K:
                    return 1
            
        return -1
