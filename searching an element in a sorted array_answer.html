class Solution{
    static int searchInSorted(int arr[], int N, int K)
    {
        
        // Your code here
        int high, low, mid;
        high=N - 1;
        low=0;
        
        while(high - low >= 0){
            mid=low + (high - low)/2;
            if(arr[mid] == K)
                return 1;
                
            else if(arr[mid] >= K)
                high = mid - 1;
                
            else
                low = mid + 1;
        }
        return -1;
        
    }
}
