class Solution:
    def majorityElement(self, A, N):
        #Your code here
        maj_index = 0
        count = 0
        for i in range(N):
            if A[maj_index] == A[i]:
                count += 1
            else:
                count -= 1
            if count == 0:
                maj_index = i
                count = 1
        
        # check whether A[maj_index] is really majority element
        count = 0
        for i in range(N):
            if A[maj_index] == A[i]:
                count += 1
            if count > N//2:
                return A[maj_index]
        
        return -1
