class Solution:
    def ValidPair(self, a, n): 
        # Your code goes here
        a.sort()
        count = 0
        left = 0
        right = n-1
        
        while left < right:
            sum = a[left] + a[right]
            if sum > 0:
                count += right-left
                right -= 1
            else:
                left += 1
        return count
