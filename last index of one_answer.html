class Solution {
    public int lastIndex( String s) {
        char[] ch = s.toCharArray();
        int ind = -1;
        for(int i=ch.length;i>0;i--){
            if(ch[i-1]=='1'){
                ind = i-1;
                return ind;
            }
        }
        return ind;
    }
}

class Solution:
    def lastIndex(self, s):
        # code here
        n = len(s)
        if '1' not in s:
            return -1
            
        res = -1
        for i in range(n-1, -1, -1):
            if s[i] == '1':
                res = i
                break
            
        return res
