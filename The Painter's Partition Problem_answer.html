class Solution:
    def minTime (self, arr, n, k):
        lo = max(arr)
        hi = sum(arr)
     
        while (lo < hi):
            mid = lo + (hi - lo) // 2
            requiredPainters = self.possible(arr, n, mid)
    
            if (requiredPainters <= k):
                hi = mid
     

            else:
                lo = mid + 1
     
        # required
        return lo
    
    def possible(self, arr, n, maxLen):
        total = 0
        numPainters = 1
     
        for i in arr:
            total += i
     
            if total > maxLen:
     
                # for next count
                total = i
                numPainters += 1
 
        
        return numPainters
