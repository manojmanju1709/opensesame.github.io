       // code here

       for(int i=5;i<=n;i++){
           if(isPrime(i)&&isPrime(i-2)){
               count++;
           }
       }
       return count;
   }
   bool isPrime(int n){
      if(n%2==0){
          return false;
      }
      for(int i=3;i<=sqrt(n);i++){
          if(n%i==0){
              return false;
          }
      }
      return true;
   }

            else:
                l.append(i)

        p=[i-2 for i in l]
