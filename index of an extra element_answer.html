class Solution{
  public:
    int findExtra(int a[], int b[], int n) {
        // add code here.
        int l = 0, r = n-1, mid;
        if (a[n-1] != b[n-2] && a[n-2] == b[n-3]) return n-1;
        while (l <= r && mid != n-1) {
            mid = (l+r)/2;
            if (a[mid] == b[mid]) l = mid + 1;
            if (a[mid] < b[mid]) {
                if (a[mid-1] == b[mid-1]) return mid;
                else r = mid-1;
            }
        }
    }
};

class Solution {
    public int findExtra(int a[], int b[], int n) {
        int start = 0;
        int end = n - 1;
        
        while (start < end) {
            int mid = start + (end - start) / 2;
            if(a[mid] == b[mid]) {
                start = mid + 1;
            } else {
                end = mid;
            }
        }
        return end;
    }
}
