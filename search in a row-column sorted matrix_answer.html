class Solution:
    
    #Function to search a given number in row-column sorted matrix.
    def search(self,matrix, n, m, x): 
        c=0
        for i in range(len(matrix)):
            a=(matrix[i])
            for j in range(len(a)):
                b=(a[j])
                if b==x:
                    c+=1
        if (c)==0:
            return 0
        elif c!=0:
            return 1
