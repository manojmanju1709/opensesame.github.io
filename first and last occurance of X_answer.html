class Solution{
    public ArrayList firstAndLast(int a[], int n, int x){
        // Code here
        ArrayList ans = new ArrayList<>();
        int first = -1;
        int last = -1;
        int low = 0,high = n-1;
        while (low<=high){
            int mid = low+(high-low)/2;
            if (a[mid]==x){
                first=mid;
                high=mid-1;
            }else if (xa[mid]){
                low=mid+1;
            }else
                high=mid-1;
        }
        ans.add(first);
        ans.add(last);
        return ans;
    }
}

class Solution: 
    def firstAndLast(self, arr, n, x): 
        # Code here
        first = -1
        second = -1
        
        start = 0
        end = n - 1
        
        while start <= end:
            mid = (start + end) // 2
            
            if arr[mid] == x and (mid == 0 or arr[mid-1] < x):
                first = mid
                break
            elif arr[mid] < x:
                start = mid + 1
            else:
                end = mid - 1
        
        start = 0
        end = n - 1
        
        while start <= end:
            mid = (start + end) // 2
            
            if arr[mid] == x and (mid == n-1 or arr[mid+1] > x):
                second = mid
                break
            elif arr[mid] > x:
                end = mid - 1
            else:
                start = mid + 1
        
        if first == -1 or second == -1:
            return [-1]
        return [first, second]
