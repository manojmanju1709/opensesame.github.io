class Solution:   
    def peakElement(self,arr, n):
        # Code here
        if n == 0: return 0
        elif n == 1: return 0
        elif n > 1:
            #take first element , map it as max element and maintain the index also
            #compare with all if no element is greater then it peak element
            #if it is less than some element than change the max element to the 
            #new element
            max_ele = arr[0]
            index = 0
            for i in range(n):
                if max_ele < arr[i] :
                    max_ele = arr[i]
                    index = i
                
                
            return index
