        if(abs(force) < 1e-12) return M;
        
        else if(force < 0) BS(L, M, n, magnets);
        else if(force > 0) BS(M, R, n, magnets);
    }

            getAnswer[i] = BS(magnets[i], magnets[i+1], n, magnets);
        }
    }

import decimal
class Solution:
    def jedi(self,a,x):
        force = 0
        for i in a:
            k = x-i
            force+=1/k
        return force    
    def bsearch(self,a,low,high):
        mid = (low + high)/2
        force = round(self.jedi(a,mid),5)
        if force == 0:
            return mid
        elif force > 0:
            return self.bsearch(a,mid,high)
        else:
            return self.bsearch(a,low,mid)
    
    def nullPoints(self, n, a, getAnswer):
        for i in range(n-1):
            getAnswer[i] = decimal.Decimal("{:.2f}".format(self.bsearch(a,a[i],a[i+1])))
